%{
	#include
	int word_count=0,char_count=0,line_count=0;
%}

%%
[A-Za-z]+ {word_count++; char_count+=strlen(yytext);}
. {char_count++;}
\n {line_count++;}
%%

int main()
{
	printf("press ctrl+c for exit\n\n");
	printf("Enter the String:");
	yylex();
	printf("\nWord Count:%d\n",word_count);
	printf("\nCharacter Count:%d\n",char_count);
	printf("\nLine Count:%d\n",line_count);
	return 0;
}

int yywrap()
{
	return 1;
}
